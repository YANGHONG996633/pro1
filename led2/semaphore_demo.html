0

⚡ vDetectTask Running
for(;;) {
  // 轮询GPIO
  if(Switch == LOW)
    osDelay(20ms) 去抖
    if(再次确认触发)
      Release(detectSem) ↑
    等物体离开...
  osDelay(10ms)
} // 回到循环顶
🎛 vControlTask Blocked
for(;;) {
  Acquire(detectSem) ⏳
  ENA = LOW 停带
  Release(uartSem) ↑
  Acquire(uartDoneSem) ⏳
  if(reply == 0x01)
    PA7=HIGH→delay→PA7=LOW
  ENA = HIGH 启带
} // 回到循环顶
📡 vUartTask Blocked
for(;;) {
  Acquire(uartSem) ⏳
  组帧 AA 55 10 ...
  HAL_UART_Transmit()
  HAL_UART_Receive() ⏳
  g_uartReply = ?
  Release(uartDoneSem) ↑
} // 回到循环顶
